USING DOCKER COMMIT
# 1. Run a new Ubuntu container
docker run -it --name ubuntucontainer ubuntu:latest

# 2. (Inside container) check git version / install if needed
git --version
apt update
apt install git -y
git --version

# 3. Exit container
exit

# 4. Commit container as new image
docker commit ubuntucontainer img-commit-1

# 5. Check image exists
docker images

# 6. Tag image for Docker Hub
docker tag img-commit-1 <dockerhub_username>/img-commit-1

# 7. Login to Docker Hub
docker login

# 8. Push image to Docker Hub
docker push <dockerhub_username>/img-commit-1

# 9. Run container from pushed image
docker run -it --name mycontainer <dockerhub_username>/img-commit-1




USING DOCKERFILE
# 1. Create project folder and go there
mkdir C:\Users\image-creation
cd C:\Users\image-creation

# 2. Create Dockerfile using Notepad
notepad Dockerfile

Dockerfile contents to paste:

FROM ubuntu:latest
RUN apt update
RUN apt install git -y
RUN apt install -y nginx
CMD ["nginx", "-g", "daemon off;"]

# 4. Build Docker image
docker build -t myubuntuimage .

optionally - check rename Dockerfile.txt to Dockerfile if not already.
ren Dockerfile.txt Dockerfile

# 5. Check image
docker images

# 6. Run container
docker run -d -p 8080:80 myubuntuimage

# 7. Open in browser
# http://localhost:8080

# 8. (Optional) Tag and push to Docker Hub
docker tag myubuntuimage <dockerhub_username>/myubuntuimage
docker login
docker push <dockerhub_username>/myubuntuimage
